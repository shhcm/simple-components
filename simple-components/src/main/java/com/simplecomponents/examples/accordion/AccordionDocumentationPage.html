Create a div with a wicket id to which we can assign the Accordion

                <!DOCTYPE html>
                <html xmlns:wicket="http://wicket.apache.org">
                    <body>
                        <!-- other content -->
                        <div wicket:id="accordion"/>
                        <!-- other content -->
                    </body>
                </html>

Create a list of AccordionItem's and the Accordion instance

Each AccordionItem holds a reference to a panel that defines the content. This panel will be displayed on a click to the corresponding accordion header.

                LinkedList<AccordionItem> accordionItems = new LinkedList<AccordionItem>();
                
                // Implement some panels with your controls/content and add them to the list.
                Panel panelA = new PanelA(AccordionPanel.CONTENT_ID);
                accordionItems.add(new AccordionItem("Panel A", contentPanel));
                Panel panelB = new PanelB(AccordionPanel.CONTENT_ID);
                accordionItems.add(new AccordionItem("Panel B ", contentPanel));
                Panel panelC = new PanelC(AccordionPanel.CONTENT_ID);
                accordionItems.add(new AccordionItem("Panel C ", contentPanel));
                
                Accordion accordion = new Accordion("accordion", accordionItems);
                add(accordion);
